public class MethodOverLoading
{
    // By changing no. of arguments
    public int add(int a , int b) {
        int sum = a+b;
        return sum ;
    }
    public int add(int a , int b , int c) {
        int sum = a+b+c;
        return sum ;
    }
    
    // By changing return datatype
    public double add(double a , double b) {
        double sum = a+b;
        return sum ;
    }
    public double add( double a , double b , double c) {
        double sum = a+b+c;
        return sum ;
    }
    
	public static void main(String[] args) {
	    MethodOverLoading obj = new MethodOverLoading();
	    
		System.out.println("sum : "+ obj.add(2,3));
		System.out.println("sum : "+ obj.add(2,3,5));
		
		System.out.println("sum : "+ obj.add(2.2,3.3));
		System.out.println("sum : "+ obj.add(2.2,3.3,5.5));
	}
}
