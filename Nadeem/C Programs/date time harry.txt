#include <stdio.h>
typedef struct time{
   int hr;
   int min;
   int sec;
}time;
void display_time(time t){
    printf("      %d    %d    %d",t.hr,t.min,t.sec);
}
int comp_time(time t1,time t2){
    if(t1.hr>t2.hr){
        return 1;
    }
    if(t1.hr<t2.hr){
        return -1;
    }if(t1.min>t2.min){
        return 1;
    }if(t2.min>t2.min){
        return 1;
    }if(t1.sec>t2.sec){
        return 1;
    }if(t1.sec>t2.sec){
        return 1;
    }
    else{
        return 0;
    }
}
