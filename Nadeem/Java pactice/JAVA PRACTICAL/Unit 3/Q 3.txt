// Parent class
class Animal
{
    void eat()
    {
        System.out.println("eating...");
    }
}

// Subclass
class Lion extends Animal
{
    @Override
    void eat()
    {
        System.out.println("eating meat...");
    }
}



public class overriding
{
	public static void main(String[] args) {
		Animal obj = new Lion();
		obj.eat();
	}
}
