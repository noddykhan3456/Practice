import java.io.*;

class employee
{
    int age;
    String name,designation;
    double salary;
    
    // Input
    void getData(int a ,String n ,String d ,double s)
    { 
        age = a;
        name = n;
        designation = d;
        salary = s;
    }
    
    // Output
    void printEmployee()
    {
		System.out.println("Age : " + age);
		System.out.println("Name : " + name);
		System.out.println("Designation : " + designation);
		System.out.println("Salart : " + salary);
    }
}

public class Main
{
	public static void main(String[] args)
	{ 
	   employee emp = new employee();
	   emp.getData(18,"Mohd Nadeem","Software Developer",50000);
	   emp.printEmployee();
	}
}
